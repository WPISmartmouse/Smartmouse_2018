290 EIGEN_STRONG_INLINE EIGEN_DEVICE_FUNC __half_raw float_to_half_rtne(float ff) {
291 #if defined(EIGEN_HAS_CUDA_FP16) && defined(EIGEN_CUDA_ARCH) && EIGEN_CUDA_ARCH >= 300
292  __half tmp_ff = __float2half(ff);
293  return *(__half_raw*)&tmp_ff;
294 
295 #elif defined(EIGEN_HAS_FP16_C)
296  __half_raw h;
297  h.x = _cvtss_sh(ff, 0);
298  return h;
299 
300 #else
301  FP32 f; f.f = ff;
302 
303  const FP32 f32infty = { 255 << 23 };
304  const FP32 f16max = { (127 + 16) << 23 };
305  const FP32 denorm_magic = { ((127 - 15) + (23 - 10) + 1) << 23 };
306  unsigned int sign_mask = 0x80000000u;
307  __half_raw o;
308  o.x = static_cast<unsigned short>(0x0u);
309 
310  unsigned int sign = f.u & sign_mask;
311  f.u ^= sign;
312 
313  // NOTE all the integer compares in this function can be safely
314  // compiled into signed compares since all operands are below
315  // 0x80000000. Important if you want fast straight SSE2 code
316  // (since there's no unsigned PCMPGTD).
317 
318  if (f.u >= f16max.u) { // result is Inf or NaN (all exponent bits set)
319  o.x = (f.u > f32infty.u) ? 0x7e00 : 0x7c00; // NaN->qNaN and Inf->Inf
320  } else { // (De)normalized number or zero
321  if (f.u < (113 << 23)) { // resulting FP16 is subnormal or zero
322  // use a magic value to align our 10 mantissa bits at the bottom of
323  // the float. as long as FP addition is round-to-nearest-even this
324  // just works.
325  f.f += denorm_magic.f;
326 
327  // and one integer subtract of the bias later, we have our final float!
328  o.x = static_cast<unsigned short>(f.u - denorm_magic.u);
329  } else {
330  unsigned int mant_odd = (f.u >> 13) & 1; // resulting mantissa is odd
331 
332  // update exponent, rounding bias part 1
333  f.u += ((unsigned int)(15 - 127) << 23) + 0xfff;
334  // rounding bias part 2
335  f.u += mant_odd;
336  // take the bits!
337  o.x = static_cast<unsigned short>(f.u >> 13);
338  }
339  }
340 
341  o.x |= static_cast<unsigned short>(sign >> 16);
342  return o;
343 #endif
344 }

346 EIGEN_STRONG_INLINE EIGEN_DEVICE_FUNC float half_to_float(__half_raw h) {
347 #if defined(EIGEN_HAS_CUDA_FP16) && defined(EIGEN_CUDA_ARCH) && EIGEN_CUDA_ARCH >= 300
348  return __half2float(h);
349 
350 #elif defined(EIGEN_HAS_FP16_C)
351  return _cvtsh_ss(h.x);
352 
353 #else
354  const FP32 magic = { 113 << 23 };
355  const unsigned int shifted_exp = 0x7c00 << 13; // exponent mask after shift
356  FP32 o;
357 
358  o.u = (h.x & 0x7fff) << 13; // exponent/mantissa bits
359  unsigned int exp = shifted_exp & o.u; // just the exponent
360  o.u += (127 - 15) << 23; // exponent adjust
361 
362  // handle exponent special cases
363  if (exp == shifted_exp) { // Inf/NaN?
364  o.u += (128 - 16) << 23; // extra exp adjust
365  } else if (exp == 0) { // Zero/Denormal?
366  o.u += 1 << 23; // extra exp adjust
367  o.f -= magic.f; // renormalize
368  }
369 
370  o.u |= (h.x & 0x8000) << 16; // sign bit
371  return o.f;
372 #endif
373 }
