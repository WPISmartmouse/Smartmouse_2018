567 template<typename Packet> EIGEN_DEVICE_FUNC inline Packet
568 pinsertfirst(const Packet& a, typename unpacket_traits<Packet>::type b)
569 {
570  // Default implementation based on pblend.
571  // It must be specialized for higher performance.
572  Selector<unpacket_traits<Packet>::size> mask;
573  mask.select[0] = true;
574  // This for loop should be optimized away by the compiler.
575  for(Index i=1; i<unpacket_traits<Packet>::size; ++i)
576  mask.select[i] = false;
577  return pblend(mask, pset1<Packet>(b), a);
578 }
