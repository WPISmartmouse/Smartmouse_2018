50 // promote_scalar_arg is an helper used in operation between an expression and a scalar, like:
51 // expression * scalar
52 // Its role is to determine how the type T of the scalar operand should be promoted given the scalar type ExprScalar of the given expression.
53 // The IsSupported template parameter must be provided by the caller as: internal::has_ReturnType<ScalarBinaryOpTraits<ExprScalar,T,op> >::value using the proper order for ExprScalar and T.
54 // Then the logic is as follows:
55 // - if the operation is natively supported as defined by IsSupported, then the scalar type is not promoted, and T is returned.
56 // - otherwise, NumTraits<ExprScalar>::Literal is returned if T is implicitly convertible to NumTraits<ExprScalar>::Literal AND that this does not imply a float to integer conversion.
57 // - otherwise, ExprScalar is returned if T is implicitly convertible to ExprScalar AND that this does not imply a float to integer conversion.
58 // - In all other cases, the promoted type is not defined, and the respective operation is thus invalid and not available (SFINAE).
59 template<typename ExprScalar,typename T, bool IsSupported>
60 struct promote_scalar_arg;
61

63 struct promote_scalar_arg<S,T,true>
64 {
65  typedef T type;
66 };

437 // However, we still need a mechanism to detect whether an expression which is evaluated multiple time
438 // has to be evaluated into a temporary.
439 // That's the purpose of this new nested_eval helper:
451 template<typename T, int n, typename PlainObject = typename plain_object_eval<T>::type> struct nested_eval
452 {
453  enum {
454  ScalarReadCost = NumTraits<typename traits<T>::Scalar>::ReadCost,
455  CoeffReadCost = evaluator<T>::CoeffReadCost, // NOTE What if an evaluator evaluate itself into a tempory?
456  // Then CoeffReadCost will be small (e.g., 1) but we still have to evaluate, especially if n>1.
457  // This situation is already taken care by the EvalBeforeNestingBit flag, which is turned ON
458  // for all evaluator creating a temporary. This flag is then propagated by the parent evaluators.
459  // Another solution could be to count the number of temps?
460  NAsInteger = n == Dynamic ? HugeCost : n,
461  CostEval = (NAsInteger+1) * ScalarReadCost + CoeffReadCost,
462  CostNoEval = NAsInteger * CoeffReadCost,
463  Evaluate = (int(evaluator<T>::Flags) & EvalBeforeNestingBit) || (int(CostEval) < int(CostNoEval))
464  };
465 
466  typedef typename conditional<Evaluate, PlainObject, typename ref_selector<T>::type>::type type;
