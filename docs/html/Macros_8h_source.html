663 // Static and dynamic alignment control
664 //
665 // The main purpose of this section is to define EIGEN_MAX_ALIGN_BYTES and EIGEN_MAX_STATIC_ALIGN_BYTES
666 // as the maximal boundary in bytes on which dynamically and statically allocated data may be alignment respectively.
667 // The values of EIGEN_MAX_ALIGN_BYTES and EIGEN_MAX_STATIC_ALIGN_BYTES can be specified by the user. If not,
668 // a default value is automatically computed based on architecture, compiler, and OS.
669 //

912 // EIGEN_SIZE_MIN_PREFER_DYNAMIC gives the min between compile-time sizes. 0 has absolute priority, followed by 1,
913 // followed by Dynamic, followed by other finite values. The reason for giving Dynamic the priority over
914 // finite values is that min(3, Dynamic) should be Dynamic, since that could be anything between 0 and 3.
915 #define EIGEN_SIZE_MIN_PREFER_DYNAMIC(a,b) (((int)a == 0 || (int)b == 0) ? 0 \
916  : ((int)a == 1 || (int)b == 1) ? 1 \
917  : ((int)a == Dynamic || (int)b == Dynamic) ? Dynamic \
918  : ((int)a <= (int)b) ? (int)a : (int)b)
919 
920 // EIGEN_SIZE_MIN_PREFER_FIXED is a variant of EIGEN_SIZE_MIN_PREFER_DYNAMIC comparing MaxSizes. The difference is that finite values
921 // now have priority over Dynamic, so that min(3, Dynamic) gives 3. Indeed, whatever the actual value is
922 // (between 0 and 3), it is not more than 3.
923 #define EIGEN_SIZE_MIN_PREFER_FIXED(a,b) (((int)a == 0 || (int)b == 0) ? 0 \
924  : ((int)a == 1 || (int)b == 1) ? 1 \
925  : ((int)a == Dynamic && (int)b == Dynamic) ? Dynamic \
926  : ((int)a == Dynamic) ? (int)b \
927  : ((int)b == Dynamic) ? (int)a \
928  : ((int)a <= (int)b) ? (int)a : (int)b)
