130  inline Index rank() const
131  {
132  using std::abs;
133  eigen_assert(m_isInitialized && "JacobiSVD is not initialized.");
134  if(m_singularValues.size()==0) return 0;
135  RealScalar premultiplied_threshold = numext::maxi<RealScalar>(m_singularValues.coeff(0) * threshold(), (std::numeric_limits<RealScalar>::min)());

137  while(i>=0 && m_singularValues.coeff(i) < premultiplied_threshold) --i;
138  return i+1;

257 template<typename RhsType, typename DstType>
258 void SVDBase<Derived>::_solve_impl(const RhsType &rhs, DstType &dst) const
259 {
260  eigen_assert(rhs.rows() == rows());
261 
262  // A = U S V^*
263  // So A^{-1} = V S^{-1} U^*
264 
265  Matrix<Scalar, Dynamic, RhsType::ColsAtCompileTime, 0, MatrixType::MaxRowsAtCompileTime, RhsType::MaxColsAtCompileTime> tmp;
266  Index l_rank = rank();
267  tmp.noalias() = m_matrixU.leftCols(l_rank).adjoint() * rhs;
268  tmp = m_singularValues.head(l_rank).asDiagonal().inverse() * tmp;
269  dst = m_matrixV.leftCols(l_rank) * tmp;
270 }
271 #endif

273 template<typename MatrixType>
274 bool SVDBase<MatrixType>::allocate(Index rows, Index cols, unsigned int computationOptions)
275 {
276  eigen_assert(rows >= 0 && cols >= 0);
277 
278  if (m_isAllocated &&
279  rows == m_rows &&
280  cols == m_cols &&
281  computationOptions == m_computationOptions)
282  {
283  return true;
284  }
285 
286  m_rows = rows;
287  m_cols = cols;
288  m_isInitialized = false;
289  m_isAllocated = true;
290  m_computationOptions = computationOptions;
291  m_computeFullU = (computationOptions & ComputeFullU) != 0;
292  m_computeThinU = (computationOptions & ComputeThinU) != 0;
293  m_computeFullV = (computationOptions & ComputeFullV) != 0;
294  m_computeThinV = (computationOptions & ComputeThinV) != 0;
295  eigen_assert(!(m_computeFullU && m_computeThinU) && "SVDBase: you can't ask for both full and thin U");
296  eigen_assert(!(m_computeFullV && m_computeThinV) && "SVDBase: you can't ask for both full and thin V");
297  eigen_assert(EIGEN_IMPLIES(m_computeThinU || m_computeThinV, MatrixType::ColsAtCompileTime==Dynamic) &&
298  "SVDBase: thin U and V are only available when your matrix has a dynamic number of columns.");
299 
300  m_diagSize = (std::min)(m_rows, m_cols);
301  m_singularValues.resize(m_diagSize);
302  if(RowsAtCompileTime==Dynamic)
303  m_matrixU.resize(m_rows, m_computeFullU ? m_rows : m_computeThinU ? m_diagSize : 0);
304  if(ColsAtCompileTime==Dynamic)
305  m_matrixV.resize(m_cols, m_computeFullV ? m_cols : m_computeThinV ? m_diagSize : 0);
306 
307  return false;
308 }
